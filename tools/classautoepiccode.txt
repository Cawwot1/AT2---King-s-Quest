# Attributes
__player_sprite = None
__enemy_sprite = None

# Constructor
def __init__(self, player_sprite, enemy_sprite):
    self.setPlayer_sprite(player_sprite)
    self.setEnemy_sprite(enemy_sprite)

# Getters
def getPlayer_sprite(self):
    return self.__player_sprite
def getEnemy_sprite(self):
    return self.__enemy_sprite

# Setters
def setPlayer_sprite(self, player_sprite):
    self.__player_sprite = player_sprite
def setEnemy_sprite(self, enemy_sprite):
    self.__enemy_sprite = enemy_sprite
