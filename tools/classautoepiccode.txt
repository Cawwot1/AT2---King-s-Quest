# Attributes
__name = None
__character_class = None
__rect = None
__direction = None
__movement_speed = None
__d = None

# Constructor
def __init__(self, name, character_class, rect, direction, movement_speed, d):
    self.setName(name)
    self.setCharacter_class(character_class)
    self.setRect(rect)
    self.setDirection(direction)
    self.setMovement_speed(movement_speed)
    self.setD(d)

# Getters
def getName(self):
    return self.__name
def getCharacter_class(self):
    return self.__character_class
def getRect(self):
    return self.__rect
def getDirection(self):
    return self.__direction
def getMovement_speed(self):
    return self.__movement_speed
def getD(self):
    return self.__d

# Setters
def setName(self, name):
    self.__name = name
def setCharacter_class(self, character_class):
    self.__character_class = character_class
def setRect(self, rect):
    self.__rect = rect
def setDirection(self, direction):
    self.__direction = direction
def setMovement_speed(self, movement_speed):
    self.__movement_speed = movement_speed
def setD(self, d):
    self.__d = d
