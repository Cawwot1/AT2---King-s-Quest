# Attributes
__attack_lv = None
__defence_lv = None
__a = None

# Constructor
def __init__(self, attack_lv, defence_lv, a):
    self.setAttack_lv(attack_lv)
    self.setDefence_lv(defence_lv)
    self.setA(a)

# Getters
def getAttack_lv(self):
    return self.__attack_lv
def getDefence_lv(self):
    return self.__defence_lv
def getA(self):
    return self.__a

# Setters
def setAttack_lv(self, attack_lv):
    self.__attack_lv = attack_lv
def setDefence_lv(self, defence_lv):
    self.__defence_lv = defence_lv
def setA(self, a):
    self.__a = a
